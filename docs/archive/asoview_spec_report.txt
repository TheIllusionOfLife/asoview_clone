3.1 施設・プラン系テーブル

-- 施設テーブル
CREATE TABLE facilities (
  facility_id    STRING(36) NOT NULL,  -- UUID
  name           STRING(200) NOT NULL,
  name_kana      STRING(200),
  description    STRING(MAX),
  prefecture_code STRING(2) NOT NULL,  -- JIS都道府県コード
  city           STRING(100),
  address        STRING(300),
  lat            FLOAT64,
  lng            FLOAT64,
  phone          STRING(20),
  website_url    STRING(500),
  main_image_url STRING(500),
  status         STRING(20) NOT NULL,  -- active/inactive/suspended
  partner_id     STRING(36) NOT NULL,  -- 施設パートナーID
  commission_rate FLOAT64 NOT NULL DEFAULT (0.15),
  created_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  updated_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
) PRIMARY KEY (facility_id);

-- プランテーブル（施設の「商品」単位）
CREATE TABLE plans (
  plan_id        STRING(36) NOT NULL,
  facility_id    STRING(36) NOT NULL,
  name           STRING(200) NOT NULL,
  description    STRING(MAX),
  price          INT64 NOT NULL,  -- 円（最低料金）
  price_unit     STRING(20) NOT NULL DEFAULT ("人"),
  duration_minutes INT64,
  min_people     INT64 NOT NULL DEFAULT (1),
  max_people     INT64,
  requires_datetime BOOL NOT NULL DEFAULT (true),
  cancellation_hours INT64 NOT NULL DEFAULT (24),  -- キャンセル可能時間前
  age_min        INT64,
  age_max        INT64,
  includes       STRING(MAX),  -- 料金に含まれるもの
  excludes       STRING(MAX),  -- 料金に含まれないもの
  status         STRING(20) NOT NULL,
  sort_order     INT64 NOT NULL DEFAULT (0),
  created_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  updated_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
) PRIMARY KEY (facility_id, plan_id),
INTERLEAVE IN PARENT facilities ON DELETE CASCADE;

-- スロット（日時枠）テーブル — 在庫管理の核心
CREATE TABLE plan_slots (
  slot_id        STRING(36) NOT NULL,
  plan_id        STRING(36) NOT NULL,
  facility_id    STRING(36) NOT NULL,
  slot_datetime  TIMESTAMP NOT NULL,
  capacity       INT64 NOT NULL,
  reserved       INT64 NOT NULL DEFAULT (0),
  held           INT64 NOT NULL DEFAULT (0),  -- 仮押さえ中
  status         STRING(20) NOT NULL DEFAULT ("open"),
  created_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  updated_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
) PRIMARY KEY (facility_id, plan_id, slot_id),
INTERLEAVE IN PARENT plans ON DELETE CASCADE;

CREATE INDEX idx_plan_slots_datetime ON plan_slots (facility_id, plan_id, slot_datetime);

3.2 ユーザー・予約系テーブル

-- ユーザーテーブル
CREATE TABLE users (
  user_id        STRING(36) NOT NULL,
  email          STRING(254) NOT NULL,
  email_verified BOOL NOT NULL DEFAULT (false),
  name           STRING(100),
  name_kana      STRING(100),
  phone          STRING(20),
  birth_date     DATE,
  gender         STRING(10),  -- male/female/other/unspecified
  prefecture_code STRING(2),
  family_type    STRING(20),  -- solo/couple/family/group
  membership_rank STRING(20) NOT NULL DEFAULT ("standard"),
  point_balance  INT64 NOT NULL DEFAULT (0),
  braze_user_id  STRING(100),  -- Braze連携ID
  authlete_sub   STRING(200),  -- Authlete Subject
  status         STRING(20) NOT NULL DEFAULT ("active"),
  created_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  updated_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
) PRIMARY KEY (user_id);

CREATE UNIQUE INDEX idx_users_email ON users (email);

-- 予約テーブル
CREATE TABLE bookings (
  booking_id     STRING(36) NOT NULL,
  user_id        STRING(36) NOT NULL,
  plan_id        STRING(36) NOT NULL,
  facility_id    STRING(36) NOT NULL,
  slot_id        STRING(36),  -- 日時指定の場合
  slot_datetime  TIMESTAMP,
  num_people     INT64 NOT NULL DEFAULT (1),
  unit_price     INT64 NOT NULL,
  subtotal       INT64 NOT NULL,
  discount_amount INT64 NOT NULL DEFAULT (0),
  point_used     INT64 NOT NULL DEFAULT (0),
  coupon_code    STRING(50),
  total_price    INT64 NOT NULL,
  status         STRING(20) NOT NULL DEFAULT ("pending"),
  -- pending / confirmed / cancelled / used / expired
  payment_method STRING(30),  -- credit_card/paypay/convenience
  payment_id     STRING(100),  -- Stripe PaymentIntent ID
  cancelled_at   TIMESTAMP,
  cancel_reason  STRING(200),
  booked_at      TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  updated_at     TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
) PRIMARY KEY (booking_id);

CREATE INDEX idx_bookings_user ON bookings (user_id, booked_at DESC);
CREATE INDEX idx_bookings_facility ON bookings (facility_id, booked_at DESC);

-- 電子チケットテーブル
CREATE TABLE tickets (
  ticket_id      STRING(36) NOT NULL,
  booking_id     STRING(36) NOT NULL,
  user_id        STRING(36) NOT NULL,
  qr_token       STRING(64) NOT NULL,  -- HMAC-SHA256署名済みトークン
  status         STRING(20) NOT NULL DEFAULT ("valid"),
  -- valid / used / expired / cancelled
  used_at        TIMESTAMP,
  scanned_by     STRING(100),  -- スキャンしたスタッフID
  issued_at      TIMESTAMP NOT NULL OPTIONS (allow_commit_timestamp=true),
  expires_at     TIMESTAMP NOT NULL,
) PRIMARY KEY (ticket_id);

CREATE UNIQUE INDEX idx_tickets_qr_token ON tickets (qr_token);
CREATE INDEX idx_tickets_booking ON tickets (booking_id);
